def find_forward(tx, start, end):
    return tx.run("""
        MATCH p = (a:Location {id:$start})-[:ROUTE*1..5]->(b:Location {id:$end})
        WHERE ALL(r IN relationships(p) WHERE r.direction = 'forward')
        WITH p,
             reduce(c=0, r IN relationships(p) | c + r.price_usd) AS cost,
             reduce(d=0, r IN relationships(p) | d + r.transit_days) AS days,
             reduce(cap=0, r IN relationships(p) | cap + r.capacity) AS total_capacity
        RETURN p, cost, days, total_capacity
        ORDER BY cost
        LIMIT 3
    """, start=start, end=end).data()
```

def find_backhaul(tx, start, end):
    return tx.run("""
        MATCH p = (a:Location {id:$end})-[:ROUTE*1..5]->(b:Location {id:$start})
        WHERE ANY(r IN relationships(p) WHERE r.direction = 'backhaul')
        WITH p,
             reduce(c=0, r IN relationships(p) | c + r.price_usd) AS cost,
             reduce(d=0, r IN relationships(p) | d + r.transit_days) AS days,
             reduce(cap=0, r IN relationships(p) | cap + r.capacity) AS total_capacity
        RETURN p, cost, days, total_capacity
        ORDER BY cost
        LIMIT 3
    """, start=start, end=end).data()
```
